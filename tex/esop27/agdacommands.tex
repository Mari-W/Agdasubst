% agdacommands.tex
% =====================================================================
%
% LaTeX macros for inline references to Agda definitions in
%   - SystemF.agda
%   - SystemFo.agda
%   - SystemFo-examples.agda
%
% Each macro renders the corresponding identifier with the right
% agda.sty highlighting class (\AgdaFunction, \AgdaDatatype,
% \AgdaInductiveConstructor, etc.) and supplies the Unicode/math symbol
% the source uses. The Agda source line(s) are included as a comment
% directly above each macro for easy navigation.
%
% Conventions
% -----------
%   * \A...           building block: a single token in math/text mode
%   * \SF...          definitions from SystemF.agda
%   * \SFO...         definitions from SystemFo.agda
%   * \SFOEx...       definitions from SystemFo-examples.agda
%   * Suffixes RR/RS/SR/SS encode the four compositionality variants;
%     R/S indicate renaming/substitution variants of a lemma.
% =====================================================================

% ---------- 3.2 Type renamings ----------
% Agda: _→ᴿ_ : Nat → Nat → Set
%       n₁ →ᴿ n₂ = Fin n₁ → Fin n₂
\providecommand{\SFRen}{\AFun{\to^{R}}}

% Agda: wkᴿ : n →ᴿ suc n
%       wkᴿ = suc
\providecommand{\SFwk}{\AFun{\mathsf{wk}^{R}}}

% Agda: idᴿ : n →ᴿ n
%       idᴿ α = α
\providecommand{\SFid}{\AFun{\mathsf{id}^{R}}}

% Agda: _∙ᴿ_ : Fin n₂ → n₁ →ᴿ n₂ → suc n₁ →ᴿ n₂
\providecommand{\SFconsR}{\AFun{\_\bullet^{R}\_}}

% Agda: _&ᴿ_ : Fin n₁ → n₁ →ᴿ n₂ → Fin n₂
\providecommand{\SFvarR}{\AFun{\_\&^{R}\_}}

% Agda: _⨟ᴿ_ : n₁ →ᴿ n₂ → n₂ →ᴿ n₃ → n₁ →ᴿ n₃
%       (ζ₁ ⨟ᴿ ζ₂) α = ζ₂ (ζ₁ α)
\providecommand{\SFcompR}{\AFun{\_\fatsemi^{R}\_}}

% Agda: _↑ᴿ : n₁ →ᴿ n₂ → suc n₁ →ᴿ suc n₂

\providecommand{\SFliftR}{\AFun{\_\uparrow^{R}}}

% Agda: _[_]ᴿ : Type n₁ → n₁ →ᴿ n₂ → Type n₂   -- type renaming traversal
\providecommand{\SFappR}{\AFun{\_[\_]^{R}}}

\providecommand{\SFSub}{\AFun{\to^{S}}}

% Agda: ⟨_⟩ : n₁ →ᴿ n₂ → n₁ →ˢ n₂

\providecommand{\SFembed}{\AFun{\langle\_\rangle}}

% Agda: _∙ˢ_ : Type n₂ → n₁ →ˢ n₂ → suc n₁ →ˢ n₂
\providecommand{\SFconsS}{\AFun{\_\bullet^{S}\_}}

% Agda: _&ˢ_ : Fin n₁ → n₁ →ˢ n₂ → Type n₂
\providecommand{\SFvarS}{\AFun{\_\&^{S}\_}}

% Agda: _↑ˢ : n₁ →ˢ n₂ → suc n₁ →ˢ suc n₂

\providecommand{\SFliftS}{\AFun{\_\uparrow^{S}}}

% Agda: _[_]ˢ : Type n₁ → n₁ →ˢ n₂ → Type n₂   -- type substitution traversal
\providecommand{\SFappS}{\AFun{\_[\_]^{S}}}

% Agda: _⨟ˢ_ : n₁ →ˢ n₂ → n₂ →ˢ n₃ → n₁ →ˢ n₃
%       (η₁ ⨟ˢ η₂) α = (η₁ α) [ η₂ ]ˢ
\providecommand{\SFcompS}{\AFun{\_\fatsemi^{S}\_}}

% ---------- 3.4 Sigma-calculus rewrite rules ----------
% Renaming β-laws
% Agda: `beta-ext-zero : zero  &ᴿ (α ∙ᴿ ζ)  ≡ α
\providecommand{\SFbetaExtZeroR}{\APost{`\beta\text{-}ext\text{-}zero}}
% Agda: `beta-ext-suc  : suc α &ᴿ (α′ ∙ᴿ ζ) ≡ α &ᴿ ζ
\providecommand{\SFbetaExtSucR}{\APost{`\beta\text{-}ext\text{-}suc}}
% Agda: `beta-id : α &ᴿ idᴿ ≡ α
\providecommand{\SFbetaIdR}{\APost{`\beta\text{-}id}}
% Agda: `beta-wk : α &ᴿ wkᴿ ≡ suc α
\providecommand{\SFbetaWkR}{\APost{`\beta\text{-}wk}}
% Agda: `beta-comp : α &ᴿ (ζ₁ ⨟ᴿ ζ₂) ≡ (α &ᴿ ζ₁) &ᴿ ζ₂
\providecommand{\SFbetaCompR}{\APost{`\beta\text{-}comp}}
% Agda: `associativity  : (ζ₁ ⨟ᴿ ζ₂) ⨟ᴿ ζ₃ ≡ ζ₁ ⨟ᴿ (ζ₂ ⨟ᴿ ζ₃)
\providecommand{\SFassocR}{\APost{`associativity}}
% Agda: `distributivity : (α ∙ᴿ ζ₁) ⨟ᴿ ζ₂ ≡ (α &ᴿ ζ₂) ∙ᴿ (ζ₁ ⨟ᴿ ζ₂)
\providecommand{\SFdistR}{\APost{`distributivity}}
% Agda: `interact : wkᴿ ⨟ᴿ (α ∙ᴿ ζ) ≡ ζ
\providecommand{\SFinteractR}{\APost{`interact}}
% Agda: `comp-idᵣ : ζ ⨟ᴿ idᴿ ≡ ζ
\providecommand{\SFcompIdrR}{\APost{`comp\text{-}id_r}}
% Agda: `comp-idₗ : idᴿ ⨟ᴿ ζ ≡ ζ
\providecommand{\SFcompIdlR}{\APost{`comp\text{-}id_l}}
% Agda: `η-id  : zero {n} ∙ᴿ wkᴿ ≡ idᴿ
\providecommand{\SFetaIdR}{\APost{`\eta\text{-}id}}
% Agda: `η-law : (zero &ᴿ ζ) ∙ᴿ (wkᴿ ⨟ᴿ ζ) ≡ ζ
\providecommand{\SFetaLawR}{\APost{`\eta\text{-}law}}

% Substitution β-laws
% Agda: beta-ext-zero : zero  &ˢ (T ∙ˢ η) ≡ T
\providecommand{\SFbetaExtZero}{\APost{\beta\text{-}ext\text{-}zero}}
% Agda: beta-ext-suc  : suc α &ˢ (T ∙ˢ η) ≡ α &ˢ η
\providecommand{\SFbetaExtSuc}{\APost{\beta\text{-}ext\text{-}suc}}
% Agda: beta-rename : α &ˢ ⟨ ζ ⟩ ≡ ` (α &ᴿ ζ)
\providecommand{\SFbetaRename}{\APost{\beta\text{-}rename}}
% Agda: beta-comp   : α &ˢ (η₁ ⨟ˢ η₂) ≡ (α &ˢ η₁) [ η₂ ]ˢ
\providecommand{\SFbetaComp}{\APost{\beta\text{-}comp}}
% Agda: beta-lift   : η ↑ˢ ≡ (` zero) ∙ˢ (η ⨟ˢ ⟨ wkᴿ ⟩)
\providecommand{\SFbetaLift}{\APost{\beta\text{-}lift}}
% Agda: associativity  : (η₁ ⨟ˢ η₂) ⨟ˢ η₃ ≡ η₁ ⨟ˢ (η₂ ⨟ˢ η₃)
\providecommand{\SFassoc}{\APost{associativity}}
% Agda: distributivity : (T ∙ˢ η₁) ⨟ˢ η₂ ≡ (T [ η₂ ]ˢ) ∙ˢ (η₁ ⨟ˢ η₂)
\providecommand{\SFdist}{\APost{distributivity}}
% Agda: interact : ⟨ wkᴿ ⟩ ⨟ˢ (T ∙ˢ η) ≡ η
\providecommand{\SFinteract}{\APost{interact}}
% Agda: comp-idᵣ : η ⨟ˢ ⟨ idᴿ ⟩ ≡ η
\providecommand{\SFcompIdr}{\APost{comp\text{-}id_r}}
% Agda: comp-idₗ : ⟨ idᴿ ⟩ ⨟ˢ η ≡ η
\providecommand{\SFcompIdl}{\APost{comp\text{-}id_l}}
% Agda: η-id  : (` zero {n}) ∙ˢ ⟨ wkᴿ ⟩ ≡ ⟨ idᴿ ⟩
\providecommand{\SFetaId}{\APost{\eta\text{-}id}}
% Agda: η-law : (zero &ˢ η) ∙ˢ (⟨ wkᴿ ⟩ ⨟ˢ η) ≡ η
\providecommand{\SFetaLaw}{\APost{\eta\text{-}law}}

% Monad / compositionality laws (type level)
% Agda: identityᵣ : T [ idᴿ ]ᴿ ≡ T
\providecommand{\SFidentity}{\APost{identity_r}}
% Agda: identityᵣˢ : T [ ⟨ idᴿ ⟩ ]ˢ ≡ T
\providecommand{\SFidentityRS}{\APost{identity_{rs}}}
% Agda: compositionalityᴿᴿ : (T [ ζ₁ ]ᴿ) [ ζ₂ ]ᴿ ≡ T [ ζ₁ ⨟ᴿ ζ₂ ]ᴿ
\providecommand{\SFcompositionalityRR}{\APost{compositionality\ARR}}
% Agda: compositionalityᴿˢ : (T [ ζ₁ ]ᴿ) [ η₂ ]ˢ ≡ T [ ⟨ ζ₁ ⟩ ⨟ˢ η₂ ]ˢ
\providecommand{\SFcompositionalityRS}{\APost{compositionality\ARS}}
% Agda: compositionalityˢᴿ : (T [ η₁ ]ˢ) [ ζ₂ ]ᴿ ≡ T [ η₁ ⨟ˢ ⟨ ζ₂ ⟩ ]ˢ
\providecommand{\SFcompositionalitySR}{\APost{compositionality\ASR}}
% Agda: compositionalityˢˢ : (T [ η₁ ]ˢ) [ η₂ ]ˢ ≡ T [ η₁ ⨟ˢ η₂ ]ˢ
\providecommand{\SFcompositionalitySS}{\APost{compositionality\ASS}}

% Coincidence
% Agda: coincidence : T [ ⟨ ζ ⟩ ]ˢ ≡ T [ ζ ]ᴿ
\providecommand{\SFcoincidence}{\APost{coincidence}}
% Agda: coincidence-comp : ⟨ ζ₁ ⟩ ⨟ˢ ⟨ ζ₂ ⟩ ≡ ⟨ ζ₁ ⨟ᴿ ζ₂ ⟩
\providecommand{\SFcoincidenceComp}{\APost{coincidence\text{-}comp}}

% Coincidence side lemmas
% Agda: coincidence-pop-ᴿ : (⟨ ζ ↑ᴿ ⟩ ⨟ˢ ((T′ [ ζ ]ᴿ) ∙ˢ ⟨ idᴿ ⟩)) ≡ ((T′ [ ζ ]ᴿ) ∙ˢ ⟨ ζ ⟩)
\providecommand{\SFcoincPopR}{\APost{coincidence\text{-}pop\text{-}{R}}}
% Agda: coincidence-pop-ᴿ-⨟ : ⟨ zero ∙ᴿ (ζ₁ ⨟ᴿ (ζ₂ ⨟ᴿ wkᴿ)) ⟩ ⨟ˢ ((T [ ζ₁ ⨟ᴿ ζ₂ ]ᴿ) ∙ˢ ⟨ idᴿ ⟩) ≡ (T [ ζ₁ ⨟ᴿ ζ₂ ]ᴿ) ∙ˢ (⟨ ζ₁ ⟩ ⨟ˢ ⟨ ζ₂ ⟩)
\providecommand{\SFcoincPopRComp}{\APost{coincidence\text{-}pop\text{-}{R}\text{-}\fatsemi}}
% Agda: coincidence-↑ᴿ-↑ˢ : ⟨ zero ∙ᴿ (ζ₁ ⨟ᴿ wkᴿ) ⟩ ⨟ˢ ((` zero) ∙ˢ (η₂ ⨟ˢ ⟨ wkᴿ ⟩)) ≡ (` zero) ∙ˢ (⟨ ζ₁ ⟩ ⨟ˢ (η₂ ⨟ˢ ⟨ wkᴿ ⟩))
\providecommand{\SFcoincLiftRS}{\APost{coincidence\text{-}\uparrow^{R}\text{-}\uparrow^{S}}}
% Agda: coincidence-pop-ˢ : ((T′ [ η₁ ⨟ˢ ⟨ ζ₂ ⟩ ]ˢ) ∙ˢ (η₁ ⨟ˢ (⟨ ζ₂ ⨟ᴿ wkᴿ ⟩ ⨟ˢ ((T′ [ η₁ ⨟ˢ ⟨ ζ₂ ⟩ ]ˢ) ∙ˢ ⟨ idᴿ ⟩)))) ≡ ((T′ [ η₁ ⨟ˢ ⟨ ζ₂ ⟩ ]ˢ) ∙ˢ (η₁ ⨟ˢ ⟨ ζ₂ ⟩))
\providecommand{\SFcoincPopS}{\APost{coincidence\text{-}pop\text{-}{S}}}

% Lift helpers used inside proofs
% Agda: lift-idᴿ : (idᴿ {n}) ↑ᴿ ≡ idᴿ
\providecommand{\SFliftIdR}{\AFun{lift\text{-}id^{R}}}
% Agda: lift-coincidence : (⟨ ζ ⟩ ↑ˢ) ≡ ⟨ ζ ↑ᴿ ⟩
\providecommand{\SFliftCoincidence}{\AFun{lift\text{-}coincidence}}

\providecommand{\SFliftCompositionalityRS}{\AFun{lift\text{-}compositionality\ARS}}
% Agda: lift-compositionalityˢᴿ : ((η₁ ↑ˢ) ⨟ˢ ⟨ ζ₂ ↑ᴿ ⟩) ≡ ((η₁ ⨟ˢ ⟨ ζ₂ ⟩) ↑ˢ)
\providecommand{\SFliftCompositionalitySR}{\AFun{lift\text{-}compositionality\ASR}}
% Agda: lift-compositionalityˢˢ : ((η₁ ↑ˢ) ⨟ˢ (η₂ ↑ˢ)) ≡ ((η₁ ⨟ˢ η₂) ↑ˢ)
\providecommand{\SFliftCompositionalitySS}{\AFun{lift\text{-}compositionality\ASS}}

% ---------- 3.9 Expression-level renamings ----------
% Agda: _∣_⇒ᴿ_ : n₁ →ᴿ n₂ → Ctx n₁ → Ctx n₂ → Set

\providecommand{\SFERen}{\AFun{\_\mid\_\Rightarrow^{R}\_}}

% Agda: Idᴿ : idᴿ ∣ Γ ⇒ᴿ Γ

\providecommand{\SFEIdR}{\AFun{\mathsf{Id}^{R}}}

%       Wkᴿ _ _ = suc
\providecommand{\SFEWkR}{\AFun{\mathsf{Wk}^{R}}}

% Agda: wkᴿ* : wkᴿ ∣ Γ ⇒ᴿ (Γ ▷*)
%       wkᴿ* _ x = suc* x
\providecommand{\SFEwkstarR}{\AFun{\mathsf{wk}^{R\ast}}}

% Agda: _,_∣_⨾ᴿ_ : ∀ ζ₁ ζ₂ → ζ₁ ∣ Γ₁ ⇒ᴿ Γ₂ → ζ₂ ∣ Γ₂ ⇒ᴿ Γ₃ → (ζ₁ ⨟ᴿ ζ₂) ∣ Γ₁ ⇒ᴿ Γ₃
\providecommand{\SFECompR}{\AFun{\_\,,\_\mid\_\bdot^{R}\_}}

% Agda: _⨾ᴿ_ : ζ₁ ∣ Γ₁ ⇒ᴿ Γ₂ → ζ₂ ∣ Γ₂ ⇒ᴿ Γ₃ → (ζ₁ ⨟ᴿ ζ₂) ∣ Γ₁ ⇒ᴿ Γ₃
%       _⨾ᴿ_ = _,_∣_⨾ᴿ_ _ _
\providecommand{\SFECompRBare}{\AFun{\_\bdot^{R}\_}}

% Agda: _∣_∙ᴿ_ : ∀ ζ → Γ₂ ∋ (T [ ζ ]ᴿ) → ζ ∣ Γ₁ ⇒ᴿ Γ₂ → ζ ∣ (Γ₁ ▷ T) ⇒ᴿ Γ₂
\providecommand{\SFEextR}{\AFun{\_\mid\_\bullet^{R}\_}}

% Agda: _∣_⇑ᴿ_ : ∀ ζ → ζ ∣ Γ₁ ⇒ᴿ Γ₂ → ∀ T → ζ ∣ (Γ₁ ▷ T) ⇒ᴿ (Γ₂ ▷ (T [ ζ ]ᴿ))
\providecommand{\SFEliftR}{\AFun{\_\mid\_\Uparrow^{R}\_}}

% Agda: _⇑ᴿ_ = _ ∣_⇑ᴿ_
\providecommand{\SFEliftRBare}{\AFun{\_\Uparrow^{R}\_}}

% Agda: _∣_↑ᴿ* : ∀ ζ → ζ ∣ Γ₁ ⇒ᴿ Γ₂ → (ζ ↑ᴿ) ∣ (Γ₁ ▷*) ⇒ᴿ (Γ₂ ▷*)
\providecommand{\SFETliftR}{\AFun{\_\mid\_\uparrow^{R\ast}}}
% Agda: ↑ᴿ*_ = _ ∣_↑ᴿ*
\providecommand{\SFETliftRBare}{\AFun{\uparrow^{R\ast}\_}}
